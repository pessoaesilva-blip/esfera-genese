// Sala const roomGeometry = new THREE.BoxGeometry(1000, 500, 1000); const floorTexture = new THREE.TextureLoader().load("https://threejs.org/examples/textures/metal.jpg"); const roomMaterial = new THREE.MeshPhongMaterial({ map: floorTexture, side: THREE.BackSide }); const room = new THREE.Mesh(roomGeometry, roomMaterial); scene.add(room); // Cúpula azul const domeGeometry = new THREE.SphereGeometry(500, 32, 32, 0, Math.PI * 2, 0, Math.PI / 2); const domeMaterial = new THREE.MeshPhongMaterial({ color: 0x3399ff, transparent: true, opacity: 0.5 }); const dome = new THREE.Mesh(domeGeometry, domeMaterial); dome.position.y = 250; scene.add(dome); // Praça central const plazaGeometry = new THREE.CircleGeometry(100, 32); const plazaMaterial = new THREE.MeshLambertMaterial({ color: 0x666666 }); const plaza = new THREE.Mesh(plazaGeometry, plazaMaterial); plaza.rotation.x = -Math.PI/2; scene.add(plaza); // Bancos for (let i = 0; i < 4; i++) { const benchGeometry = new THREE.BoxGeometry(60, 10, 20); const benchMaterial = new THREE.MeshLambertMaterial({ color: 0x999999 }); const bench = new THREE.Mesh(benchGeometry, benchMaterial); const angle = (i / 4) * Math.PI * 2; bench.position.set(Math.cos(angle) * 120, 5, Math.sin(angle) * 120); scene.add(bench); } // Entidade luminosa const entityGeometry = new THREE.SphereGeometry(10, 32, 32); const entityMaterial = new THREE.MeshBasicMaterial({ color: 0x88ccff, transparent: true, opacity: 0.7 }); const entity = new THREE.Mesh(entityGeometry, entityMaterial); entity.position.set(0, 20, 0); scene.add(entity); // 12 esferas flutuantes const spheres = []; for (let i = 0; i < 12; i++) { const angle = (i / 12) * Math.PI * 2; const x = Math.cos(angle) * 60; const z = Math.sin(angle) * 60; const sphereGeo = new THREE.SphereGeometry(5, 16, 16); const sphereMat = new THREE.MeshPhongMaterial({ color: 0xffffff }); const s = new THREE.Mesh(sphereGeo, sphereMat); s.position.set(x, 20, z); scene.add(s); spheres.push(s); }
