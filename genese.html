// Piso metálico antigo const floorTexture = new THREE.TextureLoader().load("https://threejs.org/examples/textures/metal.jpg"); floorTexture.wrapS = floorTexture.wrapT = THREE.RepeatWrapping; floorTexture.repeat.set(10, 10); const roomMaterial = new THREE.MeshPhongMaterial({ map: floorTexture, side: THREE.BackSide, shininess: 10 }); const roomGeometry = new THREE.BoxGeometry(1000, 500, 1000); const room = new THREE.Mesh(roomGeometry, roomMaterial); scene.add(room); // Cúpula azul com iluminação const domeGeometry = new THREE.SphereGeometry(500, 32, 32, 0, Math.PI * 2, 0, Math.PI / 2); const domeMaterial = new THREE.MeshPhongMaterial({ color: 0x3399ff, transparent: true, opacity: 0.4, emissive: 0x224488 }); const dome = new THREE.Mesh(domeGeometry, domeMaterial); dome.position.y = 250; scene.add(dome); // Luz azul suave da cúpula const domeLight = new THREE.PointLight(0x3399ff, 0.5, 1000); domeLight.position.set(0, 250, 0); scene.add(domeLight); // Praça central em pedra const plazaTexture = new THREE.TextureLoader().load("https://threejs.org/examples/textures/stone.jpg"); const plazaGeometry = new THREE.CircleGeometry(120, 32); const plazaMaterial = new THREE.MeshLambertMaterial({ map: plazaTexture }); const plaza = new THREE.Mesh(plazaGeometry, plazaMaterial); plaza.rotation.x = -Math.PI/2; scene.add(plaza); // Bancos mais detalhados for (let i = 0; i < 4; i++) { const benchSeat = new THREE.Mesh(new THREE.BoxGeometry(60, 5, 20), new THREE.MeshLambertMaterial({ color: 0x999999 })); const benchLeg1 = new THREE.Mesh(new THREE.CylinderGeometry(2, 2, 10, 8), new THREE.MeshLambertMaterial({ color: 0x555555 })); const benchLeg2 = benchLeg1.clone(); benchSeat.add(benchLeg1); benchSeat.add(benchLeg2); benchLeg1.position.set(-25, -5, 0); benchLeg2.position.set(25, -5, 0); const angle = (i / 4) * Math.PI * 2; benchSeat.position.set(Math.cos(angle) * 150, 5, Math.sin(angle) * 150); scene.add(benchSeat); }
